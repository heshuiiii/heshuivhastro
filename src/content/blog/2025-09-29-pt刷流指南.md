# jerry048 fork版本
```

bash <(wget -qO- https://raw.githubusercontent.com/guowanghushifu/Dedicated-Seedbox-Mod/refs/heads/main/Install.sh) \
  -u user\
  -p passwd\
  -c 10240 \
  -q 4.3.8 \
  -l v1.2.14 \
  -z



脚本参数说明
必需参数（安装 qBittorrent 时）
-u <username> - 设置用户名
-p <password> - 设置密码
-c <cache_size> - 设置缓存大小（单位：MiB）
-q <version> - 指定 qBittorrent 版本（如 4.3.9）
-l <version> - 指定 libtorrent 版本（如 v1.2.19）
-r - 安装 autoremove-torrents
-b - 安装 autobrr
-v - 安装 vertex
-o - 自定义端口（会提示输入各服务端口）

BBR 拥塞控制算法（四选一）

-x - 安装 BBRx
-y - 安装 BBRy
-z - 安装 BBRz
-3 - 安装 BBRv3

其他
-h - 显示帮助信息



bash <(wget -qO- https://raw.githubusercontent.com/guowanghushifu/Dedicated-Seedbox-Mod/refs/heads/main/Install.sh) \
  -u 用户名 \
  -p 密码 \
  -c 500 \

  -l v2.0.11 \
  -o \
  -z
```

```
cat > /etc/sysctl.d/99-racing-pt.conf << 'EOF'
# ============ 超大缓冲区（竞速专用）============
net.core.rmem_max = 536870912              # 512MB
net.core.wmem_max = 536870912              # 512MB
net.ipv4.tcp_rmem = 8192 262144 268435456  # 最大256MB
net.ipv4.tcp_wmem = 8192 262144 268435456  # 最大256MB
net.core.netdev_max_backlog = 1000000      # 100万！
net.core.optmem_max = 131072

# ============ 极限连接数 ============
net.core.somaxconn = 131072                # 最大化
net.ipv4.tcp_max_syn_backlog = 32768
net.ipv4.ip_local_port_range = 1024 65535
net.ipv4.tcp_max_tw_buckets = 5000000      # 500万

# ============ 超激进TCP（竞速核心）============
net.ipv4.tcp_congestion_control = bbrplus
net.ipv4.tcp_slow_start_after_idle = 0     # ⭐ 必须
net.ipv4.tcp_no_metrics_save = 1           # ⭐ 必须
net.ipv4.tcp_tw_reuse = 1
net.ipv4.tcp_tw_recycle = 1                # 旧内核可用
net.ipv4.tcp_fin_timeout = 5               # ⭐ 更快回收（从15降到5）
net.ipv4.tcp_orphan_retries = 1            # ⭐ 快速放弃死连接
net.ipv4.tcp_syn_retries = 2               # 快速放弃SYN
net.ipv4.tcp_synack_retries = 2

# ============ 保持连接活跃 ============
net.ipv4.tcp_keepalive_time = 300          # 5分钟（从600降低）
net.ipv4.tcp_keepalive_intvl = 5           # 快速探测
net.ipv4.tcp_keepalive_probes = 3

# ============ 窗口和缓冲优化 ============
net.ipv4.tcp_window_scaling = 1
net.ipv4.tcp_timestamps = 1
net.ipv4.tcp_sack = 1
net.ipv4.tcp_fack = 1
net.ipv4.tcp_ecn = 0                       # 禁用ECN（更激进）

# ============ 快速重传 ============
net.ipv4.tcp_fastopen = 3
net.ipv4.tcp_mtu_probing = 1
net.ipv4.tcp_base_mss = 1460

# ============ 竞速专用调优 ============
net.ipv4.tcp_autocorking = 0               # ⭐ 禁用corking，立即发送
net.ipv4.tcp_tso_win_divisor = 1           # ⭐ 最激进的TSO
net.ipv4.tcp_adv_win_scale = -2            # 更多发送缓冲
net.ipv4.tcp_moderate_rcvbuf = 1
net.ipv4.tcp_low_latency = 1               # 低延迟模式

# ============ 内存 ============
net.ipv4.tcp_mem = 1048576 2097152 4194304
net.ipv4.udp_mem = 1048576 2097152 4194304
net.core.rmem_default = 16777216
net.core.wmem_default = 16777216
vm.min_free_kbytes = 262144

# ============ 连接队列深度 ============
net.ipv4.tcp_max_orphans = 262144
net.ipv4.tcp_retries1 = 2
net.ipv4.tcp_retries2 = 5                  # 快速放弃

# ============ 零拷贝优化 ============
net.ipv4.tcp_sendfile = 1
EOF

sysctl -p /etc/sysctl.d/99-racing-pt.conf


```
